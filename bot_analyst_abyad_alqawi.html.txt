  <textarea id="inputText" placeholder="صف الإشارة هنا..."></textarea>
  <button onclick="analyze()">حلل الإشارة</button>
  <div id="response"></div>

  <script>
    function analyze() {
      const input = document.getElementById('inputText').value.trim();
      const responseDiv = document.getElementById('response');
      if (!input) {
        responseDiv.textContent = 'يرجى كتابة وصف للإشارة أولاً.';
        return;
      }
      // تحليل بسيط تجريبي
      let response = 'تحليل تقريبي:\n';
      if (input.includes('جرن')) {
        response += 'وجود جرن يشير إلى احتمال وجود مكان لتجميع الماء أو الإشارات الأثرية.\n';
      }
      if (input.includes('سيال')) {
        response += 'السيال قد يدل على اتجاه فتحات أو ممرات مخفية.\n';
      }
      if (input.includes('باب')) {
        response += 'الباب الصخري قد يكون مدخل لمغارة أو مكان دفين.\n';
      }
      if (response === 'تحليل تقريبي:\n') {
        response = 'عذرًا، لا يوجد إشارات واضحة في الوصف المقدم. يرجى إضافة تفاصيل أكثر.';
      }
      responseDiv.textContent = response;
    }
  </script>
